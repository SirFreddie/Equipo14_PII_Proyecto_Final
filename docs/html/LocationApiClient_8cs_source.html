72 public async Task<Location> GetLocationAsync(
73 string address,
74 string city = "Montevideo",
75 string department = "Montevideo",
76 string country = "Uruguay")
77 {
78 var parameters = new Dictionary<string, string>
79 {
80 { "address", address },
81 { "city", city },
82 { "department", department },
83 { "country", country },
84 };
85
86 var requestUri = GetUri(LocationUrl, parameters);
87
88 Location result;
89
90 using (var response = await this.client.GetAsync(requestUri).ConfigureAwait(false))
91 {
92 response.EnsureSuccessStatusCode();
93 string content = await response.Content.ReadAsStringAsync().ConfigureAwait(false);
94 result = JsonSerializer.Deserialize<Location>(
95 content,
96 new JsonSerializerOptions() { PropertyNameCaseInsensitive = true });
97 }
98
99 return result;
100 }

200 public async Task DownloadMapAsync(double latitude, double longitude, string path, int zoomLevel = 15)

302 private static Uri GetUri(string baseUrl, IDictionary<string, string> parameters)
303 {
304 return new Uri(
305 string.Format(
306 CultureInfo.InvariantCulture,
307 "{0}?{1}",
308 baseUrl,
309 string.Join(
310 "&",
311 parameters.Select(kvp =>
312 string.Format(
313 CultureInfo.InvariantCulture,
314 "{0}={1}",
315 kvp.Key,
316 HttpUtility.UrlEncode(kvp.Value))))));
317 }
